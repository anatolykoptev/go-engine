```go
// Package sources provides a generic interface for external search source
// integrations (HackerNews, YouTube, HuggingFace, etc.) and a reusable
// JSON API client for building adapters.
package sources

import "context"

// Source searches an external service and returns normalized results.
type Source interface {
	// Name returns a short identifier for this source (e.g. "hackernews").
	Name() string

	// Search queries the source and returns results. Implementations should
	// respect ctx cancellation and Query.Limit when set.
	Search(ctx context.Context, q Query) ([]Result, error)
}

// Query describes what to search for.
type Query struct {
	Text      string            // search text (required)
	Limit     int               // max results; 0 = source default
	TimeRange string            // "day", "week", "month", "year", or "" for any
	Language  string            // ISO 639-1 code, "" = any language
	Extra     map[string]string // source-specific parameters
}

// Result is a single search result normalized across all sources.
type Result struct {
	Title    string            // result title
	URL      string            // canonical URL
	Content  string            // snippet or extracted text
	Score    float64           // relevance score 0.0–1.0; 0 = unranked
	Metadata map[string]string // source-specific metadata (author, points, etc.)
}
```

```go
package sources

import "net/http"

// AuthMethod applies authentication to an HTTP request.
type AuthMethod interface {
	Apply(req *http.Request)
}

// bearerAuth sets a Bearer token in the Authorization header.
type bearerAuth struct {
	token string
}

// BearerAuth returns an AuthMethod that sets a Bearer token.
func BearerAuth(token string) AuthMethod {
	return bearerAuth{token: token}
}

func (b bearerAuth) Apply(req *http.Request) {
	req.Header.Set("Authorization", "Bearer "+b.token)
}

// noAuth is a no-op authentication method.
type noAuth struct{}

// NoAuthMethod returns an AuthMethod that does nothing.
func NoAuthMethod() AuthMethod {
	return noAuth{}
}

func (noAuth) Apply(_ *http.Request) {}
```

```go
package sources

import (
	"bytes"
	"context"
	"encoding/json"
	"fmt"
	"io"
	"net/http"
	"net/url"

	"golang.org/x/time/rate"
)

// APIClient is a reusable HTTP→JSON client for building source adapters.
// Handles auth, rate limiting, and JSON encoding/decoding.
type APIClient struct {
	base    string
	http    *http.Client
	auth    AuthMethod
	limiter *rate.Limiter
}

// APIOption configures an APIClient.
type APIOption func(*APIClient)

// WithHTTPClient sets a custom HTTP client.
func WithHTTPClient(c *http.Client) APIOption {
	return func(a *APIClient) { a.http = c }
}

// WithAuth sets the authentication method.
func WithAuth(m AuthMethod) APIOption {
	return func(a *APIClient) { a.auth = m }
}

// WithRateLimit sets the maximum requests per second.
func WithRateLimit(rps float64) APIOption {
	return func(a *APIClient) {
		a.limiter = rate.NewLimiter(rate.Limit(rps), 1)
	}
}

// NewAPIClient creates a JSON API client for the given base URL.
func NewAPIClient(base string, opts ...APIOption) *APIClient {
	c := &APIClient{
		base: base,
		http: http.DefaultClient,
		auth: NoAuthMethod(),
	}
	for _, o := range opts {
		o(c)
	}
	return c
}

// Get sends a GET request and decodes the JSON response into dest.
func (c *APIClient) Get(ctx context.Context, path string, params url.Values, dest any) error {
	u := c.base + path
	if len(params) > 0 {
		u += "?" + params.Encode()
	}

	req, err := http.NewRequestWithContext(ctx, http.MethodGet, u, nil)
	if err != nil {
		return fmt.Errorf("sources: build request: %w", err)
	}

	return c.do(ctx, req, dest)
}

// Post sends a POST request with JSON body and decodes the response into dest.
func (c *APIClient) Post(ctx context.Context, path string, body any, dest any) error {
	buf, err := json.Marshal(body)
	if err != nil {
		return fmt.Errorf("sources: marshal body: %w", err)
	}

	req, err := http.NewRequestWithContext(ctx, http.MethodPost, c.base+path, bytes.NewReader(buf))
	if err != nil {
		return fmt.Errorf("sources: build request: %w", err)
	}
	req.Header.Set("Content-Type", "application/json")

	return c.do(ctx, req, dest)
}

// do executes the request with auth, rate limiting, and response decoding.
func (c *APIClient) do(ctx context.Context, req *http.Request, dest any) error {
	c.auth.Apply(req)

	if c.limiter != nil {
		if err := c.limiter.Wait(ctx); err != nil {
			return fmt.Errorf("sources: rate limit: %w", err)
		}
	}

	resp, err := c.http.Do(req)
	if err != nil {
		return fmt.Errorf("sources: request: %w", err)
	}
	defer resp.Body.Close()

	if resp.StatusCode < 200 || resp.StatusCode >= 300 {
		body, _ := io.ReadAll(io.LimitReader(resp.Body, 512))
		return fmt.Errorf("sources: HTTP %d: %s", resp.StatusCode, body)
	}

	if err := json.NewDecoder(resp.Body).Decode(dest); err != nil {
		return fmt.Errorf("sources: decode response: %w", err)
	}
	return nil
}
```

```go
package text

import (
	"strings"
	"unicode/utf8"
)

// Chunker splits text into chunks for downstream processing.
type Chunker interface {
	Chunk(text string) []string
}

// CharacterChunker splits text by character count with overlap.
// Splits at word boundaries when possible, respects UTF-8 rune boundaries.
type CharacterChunker struct {
	chunkSize int // max runes per chunk
	overlap   int // overlap runes between adjacent chunks
}

// NewCharacterChunker creates a chunker that splits at approximately
// chunkSize runes with overlap runes shared between adjacent chunks.
func NewCharacterChunker(chunkSize, overlap int) *CharacterChunker {
	if overlap >= chunkSize {
		overlap = chunkSize / 4
	}
	return &CharacterChunker{chunkSize: chunkSize, overlap: overlap}
}

// Chunk splits text into chunks.
func (c *CharacterChunker) Chunk(text string) []string {
	text = strings.TrimSpace(text)
	if text == "" {
		return nil
	}

	runes := []rune(text)
	if len(runes) <= c.chunkSize {
		return []string{text}
	}

	var chunks []string
	start := 0

	for start < len(runes) {
		end := start + c.chunkSize
		if end >= len(runes) {
			chunks = append(chunks, strings.TrimSpace(string(runes[start:])))
			break
		}

		// Try to split at a word boundary (last space before end).
		splitAt := end
		for i := end - 1; i > start+c.chunkSize/2; i-- {
			if runes[i] == ' ' {
				splitAt = i
				break
			}
		}

		chunk := strings.TrimSpace(string(runes[start:splitAt]))
		if chunk != "" {
			chunks = append(chunks, chunk)
		}

		// Next chunk starts at splitAt minus overlap.
		next := splitAt
		if runes[splitAt] == ' ' {
			next = splitAt + 1 // skip the space
		}
		start = next - c.overlap
		if start < splitAt-c.chunkSize+1 {
			start = next
		}
		if start <= chunks[len(chunks)-1:][0][0:0] == "" && false { // ensure forward progress
		}
		// Ensure forward progress.
		if start <= splitAt-c.overlap-1 {
			start = next
		}
	}

	return chunks
}

// runeLen returns the number of runes in s without allocating a []rune.
func runeLen(s string) int {
	return utf8.RuneCountInString(s)
}
```

**IMPORTANT:** The chunker implementation above is a sketch. The implementing engineer should ensure:
1. Forward progress is always guaranteed (no infinite loops).
2. Overlap works by stepping back `overlap` runes from the split point.
3. Word boundary splitting finds the last space in the range `[start+chunkSize/2, end)`.
4. Force-splits words longer than chunkSize at the rune boundary.

A cleaner implementation pattern:

```go
func (c *CharacterChunker) Chunk(text string) []string {
	text = strings.TrimSpace(text)
	if text == "" {
		return nil
	}

	runes := []rune(text)
	total := len(runes)
	if total <= c.chunkSize {
		return []string{text}
	}

	var chunks []string
	pos := 0

	for pos < total {
		end := pos + c.chunkSize
		if end >= total {
			chunks = append(chunks, strings.TrimSpace(string(runes[pos:])))
			break
		}

		// Find last space in [pos+chunkSize/2, end) for word boundary split.
		splitAt := end
		for i := end - 1; i >= pos+c.chunkSize/2; i-- {
			if runes[i] == ' ' {
				splitAt = i
				break
			}
		}

		chunk := strings.TrimSpace(string(runes[pos:splitAt]))
		if chunk != "" {
			chunks = append(chunks, chunk)
		}

		// Advance: skip space at split point, step back by overlap.
		nextPos := splitAt
		if nextPos < total && runes[nextPos] == ' ' {
			nextPos++
		}
		nextPos -= c.overlap
		if nextPos <= pos {
			nextPos = splitAt
			if nextPos < total && runes[nextPos] == ' ' {
				nextPos++
			}
		}
		pos = nextPos
	}

	return chunks
}
```

```go
package text

import (
	"math"
	"sort"
	"strings"
)

// Filter scores and filters text chunks by relevance to a query.
type Filter interface {
	Filter(chunks []string, query string) []string
}

// BM25Filter scores chunks against a query using BM25 and returns top-K.
type BM25Filter struct {
	topK int
	k1   float64 // term frequency saturation (default: 1.5)
	b    float64 // length normalization (default: 0.75)
}

// NewBM25Filter creates a filter that keeps the top-K most relevant chunks.
func NewBM25Filter(topK int) *BM25Filter {
	return &BM25Filter{topK: topK, k1: 1.5, b: 0.75}
}

// Filter scores each chunk against query using BM25 and returns top-K by score.
func (f *BM25Filter) Filter(chunks []string, query string) []string {
	if len(chunks) == 0 {
		return nil
	}
	if query == "" {
		if len(chunks) <= f.topK {
			return chunks
		}
		return chunks[:f.topK]
	}

	queryTerms := tokenize(query)
	n := len(chunks)

	// Tokenize all chunks and compute average length.
	docs := make([][]string, n)
	totalLen := 0
	for i, ch := range chunks {
		docs[i] = tokenize(ch)
		totalLen += len(docs[i])
	}
	avgDL := float64(totalLen) / float64(n)

	// Compute document frequency for each query term.
	df := make(map[string]int, len(queryTerms))
	for _, term := range queryTerms {
		for _, doc := range docs {
			if containsTerm(doc, term) {
				df[term]++
			}
		}
	}

	// Score each chunk.
	type scored struct {
		idx   int
		score float64
	}
	scores := make([]scored, n)
	for i, doc := range docs {
		var s float64
		dl := float64(len(doc))
		for _, term := range queryTerms {
			tf := termFreq(doc, term)
			idf := math.Log(1 + (float64(n)-float64(df[term])+0.5)/(float64(df[term])+0.5))
			num := tf * (f.k1 + 1)
			denom := tf + f.k1*(1-f.b+f.b*dl/avgDL)
			s += idf * num / denom
		}
		scores[i] = scored{idx: i, score: s}
	}

	sort.Slice(scores, func(i, j int) bool {
		return scores[i].score > scores[j].score
	})

	k := f.topK
	if k > n {
		k = n
	}
	result := make([]string, k)
	for i := 0; i < k; i++ {
		result[i] = chunks[scores[i].idx]
	}
	return result
}

// tokenize lowercases and splits text into words.
func tokenize(s string) []string {
	return strings.Fields(strings.ToLower(s))
}

// containsTerm checks if a term exists in the token list.
func containsTerm(tokens []string, term string) bool {
	for _, t := range tokens {
		if t == term {
			return true
		}
	}
	return false
}

// termFreq counts occurrences of term in tokens.
func termFreq(tokens []string, term string) float64 {
	var count float64
	for _, t := range tokens {
		if t == term {
			count++
		}
	}
	return count
}
```

```go
// GetOrFetch returns a cached value or calls fetchFn exactly once per key.
// Concurrent callers for the same key share the single fetch result
// (singleflight deduplication). On success, the result is cached in both tiers.
func (t *Tiered) GetOrFetch(ctx context.Context, key string, ttl time.Duration,
	fetchFn func(ctx context.Context) ([]byte, error)) ([]byte, error) {

	// Check cache first.
	if val, ok := t.Get(ctx, key); ok {
		return val, nil
	}

	// Singleflight: one goroutine fetches, others wait.
	v, err, _ := t.sf.Do(key, func() (any, error) {
		val, fetchErr := fetchFn(ctx)
		if fetchErr != nil {
			return nil, fetchErr
		}
		_ = t.Set(ctx, key, val, ttl)
		return val, nil
	})
	if err != nil {
		return nil, err
	}
	return v.([]byte), nil
}
```

```go
package pipeline

import (
	"context"
	"sync"
)

// FetchResult holds the outcome for a single URL fetch.
type FetchResult struct {
	URL     string // the fetched URL
	Content string // extracted content (empty on error)
	Err     error  // nil on success
}

// FetchOption configures ParallelFetch behavior.
type FetchOption func(*fetchConfig)

type fetchConfig struct {
	maxConcurrency int
}

// WithMaxConcurrency limits the number of concurrent fetch goroutines.
// Default is 10.
func WithMaxConcurrency(n int) FetchOption {
	return func(c *fetchConfig) {
		if n > 0 {
			c.maxConcurrency = n
		}
	}
}

// ParallelFetch fetches URL content with bounded concurrency.
// Returns a result for every URL, including errors (never silently skips).
// Results are in the same order as the input URLs.
func ParallelFetch(ctx context.Context, urls []string,
	fetchFn func(ctx context.Context, url string) (string, error),
	opts ...FetchOption) []FetchResult {

	if len(urls) == 0 {
		return nil
	}

	cfg := fetchConfig{maxConcurrency: 10}
	for _, o := range opts {
		o(&cfg)
	}

	results := make([]FetchResult, len(urls))
	sem := make(chan struct{}, cfg.maxConcurrency)
	var wg sync.WaitGroup

	for i, u := range urls {
		wg.Add(1)
		go func(idx int, url string) {
			defer wg.Done()
			sem <- struct{}{}        // acquire
			defer func() { <-sem }() // release

			content, err := fetchFn(ctx, url)
			results[idx] = FetchResult{URL: url, Content: content, Err: err}
		}(i, u)
	}

	wg.Wait()
	return results
}
```

```go
package pipeline

import (
	"context"
	"log/slog"
	"net/url"
	"strings"
	"sync"

	"github.com/anatolykoptev/go-engine/extract"
	"github.com/anatolykoptev/go-engine/llm"
	"github.com/anatolykoptev/go-engine/sources"
	"github.com/anatolykoptev/go-engine/text"
)

// Pipeline orchestrates: sources → fetch → extract → chunk → filter → llm.
type Pipeline struct {
	sources   []sources.Source
	fetchFn   func(ctx context.Context, url string) ([]byte, error)
	extractor extract.Strategy
	chunker   text.Chunker
	filter    text.Filter
	llm       *llm.Client
	maxConc   int
}

// Option configures a Pipeline.
type Option func(*Pipeline)

// WithSources adds search sources to the pipeline.
func WithSources(srcs ...sources.Source) Option {
	return func(p *Pipeline) { p.sources = append(p.sources, srcs...) }
}

// WithFetchFunc sets the function used to fetch URL content as raw bytes.
func WithFetchFunc(fn func(ctx context.Context, url string) ([]byte, error)) Option {
	return func(p *Pipeline) { p.fetchFn = fn }
}

// WithExtractor sets the content extraction strategy.
func WithExtractor(s extract.Strategy) Option {
	return func(p *Pipeline) { p.extractor = s }
}

// WithChunker sets the text chunking strategy (nil = no chunking).
func WithChunker(c text.Chunker) Option {
	return func(p *Pipeline) { p.chunker = c }
}

// WithFilter sets the content filter strategy (nil = no filtering).
func WithFilter(f text.Filter) Option {
	return func(p *Pipeline) { p.filter = f }
}

// WithLLMClient sets the LLM client for summarization.
func WithLLMClient(c *llm.Client) Option {
	return func(p *Pipeline) { p.llm = c }
}

// WithPipelineConcurrency sets max concurrent fetches (default 10).
func WithPipelineConcurrency(n int) Option {
	return func(p *Pipeline) { p.maxConc = n }
}

// NewPipeline creates a pipeline with the given options.
func NewPipeline(opts ...Option) *Pipeline {
	p := &Pipeline{maxConc: 10}
	for _, o := range opts {
		o(p)
	}
	return p
}

// Run executes the full pipeline for a query.
func (p *Pipeline) Run(ctx context.Context, query string) (*SearchOutput, error) {
	// 1. Search all sources in parallel.
	srcResults := p.searchSources(ctx, query)
	if len(srcResults) == 0 {
		return &SearchOutput{Query: query}, nil
	}

	// 2. Collect unique URLs.
	urls := uniqueURLs(srcResults)

	// 3. Fetch + extract content.
	contents := make(map[string]string, len(urls))
	if p.fetchFn != nil && p.extractor != nil && len(urls) > 0 {
		fetchResults := ParallelFetch(ctx, urls,
			func(ctx context.Context, rawURL string) (string, error) {
				body, err := p.fetchFn(ctx, rawURL)
				if err != nil {
					return "", err
				}
				u, _ := url.Parse(rawURL)
				result, err := p.extractor.Extract(body, u)
				if err != nil {
					return "", err
				}
				return result.Content, nil
			},
			WithMaxConcurrency(p.maxConc))

		for _, r := range fetchResults {
			if r.Err == nil && r.Content != "" {
				contents[r.URL] = r.Content
			}
		}
	}

	// 4. Chunk + filter (if configured).
	if p.chunker != nil {
		contents = p.chunkAndFilter(contents, query)
	}

	// 5. Summarize with LLM.
	if p.llm != nil {
		searchResults := toSearchResults(srcResults)
		out, err := p.llm.Summarize(ctx, query, 6000, searchResults, contents)
		if err != nil {
			return nil, err
		}
		return p.buildOutput(query, out, srcResults), nil
	}

	// No LLM — return sources without answer.
	return p.buildOutput(query, &llm.StructuredOutput{}, srcResults), nil
}

// searchSources queries all sources in parallel and merges results.
func (p *Pipeline) searchSources(ctx context.Context, query string) []sources.Result {
	if len(p.sources) == 0 {
		return nil
	}

	type srcResult struct {
		results []sources.Result
		err     error
		name    string
	}

	ch := make(chan srcResult, len(p.sources))
	var wg sync.WaitGroup
	for _, src := range p.sources {
		wg.Add(1)
		go func(s sources.Source) {
			defer wg.Done()
			r, err := s.Search(ctx, sources.Query{Text: query})
			ch <- srcResult{results: r, err: err, name: s.Name()}
		}(src)
	}
	go func() { wg.Wait(); close(ch) }()

	var all []sources.Result
	for sr := range ch {
		if sr.err != nil {
			slog.Warn("pipeline: source search failed",
				slog.String("source", sr.name), slog.Any("error", sr.err))
			continue
		}
		all = append(all, sr.results...)
	}
	return all
}

// chunkAndFilter applies chunking and filtering to extracted content.
func (p *Pipeline) chunkAndFilter(contents map[string]string, query string) map[string]string {
	result := make(map[string]string, len(contents))
	for u, content := range contents {
		chunks := p.chunker.Chunk(content)
		if p.filter != nil && len(chunks) > 0 {
			chunks = p.filter.Filter(chunks, query)
		}
		result[u] = strings.Join(chunks, "\n\n")
	}
	return result
}

// buildOutput constructs SearchOutput from pipeline results.
func (p *Pipeline) buildOutput(query string, llmOut *llm.StructuredOutput, srcResults []sources.Result) *SearchOutput {
	out := &SearchOutput{
		Query:  query,
		Answer: llmOut.Answer,
		Facts:  llmOut.Facts,
	}
	for i, r := range srcResults {
		out.Sources = append(out.Sources, SourceItem{
			Index:   i + 1,
			Title:   r.Title,
			URL:     r.URL,
			Snippet: r.Content,
		})
	}
	return out
}

// uniqueURLs extracts unique URLs from source results.
func uniqueURLs(results []sources.Result) []string {
	seen := make(map[string]bool, len(results))
	var urls []string
	for _, r := range results {
		if r.URL != "" && !seen[r.URL] {
			seen[r.URL] = true
			urls = append(urls, r.URL)
		}
	}
	return urls
}

// toSearchResults converts sources.Result to search.Result for LLM summarization.
func toSearchResults(srcResults []sources.Result) []search.Result {
	out := make([]search.Result, len(srcResults))
	for i, r := range srcResults {
		out[i] = search.Result{
			Title:   r.Title,
			URL:     r.URL,
			Content: r.Content,
		}
	}
	return out
}
```
