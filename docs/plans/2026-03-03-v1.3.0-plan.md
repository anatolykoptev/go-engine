func (c *Client) CompleteWithSystem(ctx context.Context, system, prompt string) (string, error) {
	if c.metrics != nil {
		c.metrics.Incr("llm_calls")
	}

	raw, err := c.kit.Complete(ctx, system, prompt,
		kitllm.WithChatTemperature(c.temperature),
		kitllm.WithChatMaxTokens(c.maxTokens),
	)
	if err != nil {
		if c.metrics != nil {
			c.metrics.Incr("llm_errors")
		}
		return "", err
	}

	return stripFences(raw), nil
}

package extract

import (
	"bytes"

	"golang.org/x/net/html"
	"golang.org/x/net/html/atom"
)

// largeBodyThreshold is the size above which StripScriptStyle should be
// called before DOM parsing to reduce memory.
const largeBodyThreshold = 500 * 1024 // 500KB

// StripScriptStyle removes <script> and <style> elements from raw HTML
// using the tokenizer (no full DOM parse). Returns cleaned HTML bytes.
func StripScriptStyle(body []byte) []byte {
	if len(body) == 0 {
		return body
	}

	var buf bytes.Buffer
	buf.Grow(len(body))
	z := html.NewTokenizer(bytes.NewReader(body))

	skip := false
	for {
		tt := z.Next()
		if tt == html.ErrorToken {
			break
		}

		tn, _ := z.TagName()
		tag := atom.Lookup(tn)

		switch tt {
		case html.StartTagToken:
			if tag == atom.Script || tag == atom.Style {
				skip = true
				continue
			}
		case html.EndTagToken:
			if tag == atom.Script || tag == atom.Style {
				skip = false
				continue
			}
		}

		if !skip {
			buf.Write(z.Raw())
		}
	}

	return buf.Bytes()
}
```

func (e *Extractor) Extract(ctx context.Context, body []byte, pageURL *url.URL) (*Result, error) {
	// Tier 1: go-trafilatura (best accuracy, 3-tier internal fallback).
	if result, err := e.extractTrafilatura(body, pageURL); err == nil && result.Content != "" {

}

func (e *Extractor) extractTrafilatura(body []byte, pageURL *url.URL) (*Result, error) {
	result, err := trafilatura.Extract(bytes.NewReader(body), trafilatura.Options{
		OriginalURL:     pageURL,
		EnableFallback:  true,
		Focus:           trafilatura.FavorRecall,
		ExcludeComments: true,
	})
	if err != nil {
		return nil, err
	}

	text := result.ContentText
	var content string

	switch e.format {
	case FormatMarkdown:
		if result.ContentNode != nil {
			var htmlBuf bytes.Buffer
			if renderErr := html.Render(&htmlBuf, result.ContentNode); renderErr == nil {
				if md, mdErr := htmltomarkdown.ConvertString(htmlBuf.String()); mdErr == nil && strings.TrimSpace(md) != "" {
					content = md
				}
			}
		}
		if content == "" {
			content = text // fallback to text if markdown conversion fails
		}
	case FormatHTML:
		if result.ContentNode != nil {
			var htmlBuf bytes.Buffer
			if renderErr := html.Render(&htmlBuf, result.ContentNode); renderErr == nil {
				content = htmlBuf.String()
			}
		}
		if content == "" {
			content = text
		}
	default: // FormatText
		// Use markdown as text if available (trafilatura's current behavior).
		if result.ContentNode != nil {
			var htmlBuf bytes.Buffer
			if renderErr := html.Render(&htmlBuf, result.ContentNode); renderErr == nil {
				if md, mdErr := htmltomarkdown.ConvertString(htmlBuf.String()); mdErr == nil && strings.TrimSpace(md) != "" {
					content = md
				}
			}
		}
		if content == "" {
			content = text
		}
	}

	content = strings.TrimSpace(content)

	return &Result{
		Title:   result.Metadata.Title,
		Content: e.truncate(content),
		Format:  e.format,
	}, nil
}

func (e *Extractor) extractGoquery(body []byte) (*Result, error) {
	doc, err := goquery.NewDocumentFromReader(strings.NewReader(string(body)))
	if err != nil {
		return nil, err
	}

	// Extract title.
	title := doc.Find("title").First().Text()
	if title == "" {
		doc.Find(`meta[property="og:title"]`).Each(func(_ int, s *goquery.Selection) {
			if title == "" {
				title, _ = s.Attr("content")
			}
		})
	}

	// Remove boilerplate elements.
	doc.Find(removeSelectors).Each(func(_ int, s *goquery.Selection) {
		s.Remove()
	})

	// Find main content container.
	contentSel := doc.Find(contentSelectors).First()
	if contentSel.Length() == 0 {
		contentSel = doc.Find("body")
	}

	var content string
	switch e.format {
	case FormatHTML:
		stripAttributes(doc)
		content, _ = contentSel.Html()
		content = strings.TrimSpace(content)
	case FormatMarkdown:
		html, _ := contentSel.Html()
		if md, err := htmltomarkdown.ConvertString(html); err == nil {
			content = strings.TrimSpace(md)
		}
	default: // FormatText
		content = contentSel.Text()
		content = strings.TrimSpace(content)
		content = reWhitespace.ReplaceAllString(content, " ")
		content = cleanLines(content)
	}

	return &Result{
		Title:   title,
		Content: e.truncate(content),
		Format:  e.format,
	}, nil
}
```

```go
func (e *Extractor) Extract(ctx context.Context, body []byte, pageURL *url.URL) (*Result, error) {
	// Pre-strip scripts/styles for large bodies.
	if len(body) > largeBodyThreshold {
		body = StripScriptStyle(body)
	}

	// Tier 1: go-trafilatura.
	if result, err := e.extractTrafilatura(body, pageURL); err == nil && result.Content != "" {
		if len(result.Content) >= e.minExtractChars || e.llmFallback == nil {
			return result, nil
		}
	}

	// Tier 2: goquery.
	if result, err := e.extractGoquery(body); err == nil && result.Content != "" {
		if len(result.Content) >= e.minExtractChars || e.llmFallback == nil {
			return result, nil
		}
	}

	// Tier 3: regex.
	result, err := e.extractRegex(body)
	if err != nil {
		return result, err
	}

	// LLM fallback: if content is still thin and fallback is configured.
	if e.llmFallback != nil && len(result.Content) < e.minExtractChars {
		llmContent, llmErr := e.llmFallback(ctx, string(body))
		if llmErr == nil && llmContent != "" {
			result.Content = e.truncate(llmContent)
		}
	}

	return result, nil
}
```

**IMPORTANT:** The `largeBodyThreshold` constant and `StripScriptStyle` function are from Task 3 (cleanup.go). The LLM fallback receives the original HTML (after script stripping if body was large). This is correct — the LLM can handle HTML, and the tiers already stripped most noise.
